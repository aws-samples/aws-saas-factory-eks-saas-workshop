apiVersion: apps/v1

  name: tenant-registration
spec:
  replicas: 1
  selector:
    matchLabels:
      app: tenant-registration
  template:
    metadata:
      labels:
        app: tenant-registration
    spec:
      containers:
        - name: tenant-registration
          image: $TENANTREGISTRATIONECR:latest
          imagePullPolicy: Always

              value: "$REGION"

              value: "COGNITO_USER_POOL_ID"

              value: "COGNITO_CLIENT_ID"
            - name: COGNITO_REGION
              value: "$REGION"
            - name: TENANT_TABLE_NAME
              value: "$TENANT_TABLE_NAME"

              value: "$TENANT_STACK_MAPPING_TABLE_NAME"

apiVersion: v1
kind: Service
metadata:
  name: tenant-registration-service

  selector:
    app: tenant-registration
  ports:
    - name: http
      protocol: TCP
      port: 80
      targetPort: 3000
  type: NodePort

  name: tenant-registration-service-ingress
  annotations:

    nginx.org/mergeable-ingress-type: "minion"

  rules:
    - host: $ELBURL
      http:
        paths:
          - backend:
              service:
                name: tenant-registration-service
                port:
                  number: 80
            path: /api/registration
            pathType: Prefix
